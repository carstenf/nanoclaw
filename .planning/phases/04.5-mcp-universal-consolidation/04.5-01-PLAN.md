    Step 5 — write `createSession()` helper that combines Pattern 1 (lines 234-256) with Pitfall-8 wrapping (move from old lines 205-213 into per-session tool handler closure):
    ```typescript
    async function createSession(): Promise<Session> {
      const mcp = new McpServer({ name: 'nanoclaw-voice', version: '1.0.0' });
      // Register every tool via shared registry, wrapping Pitfall-8 synthetic IDs + TOOL_META schema.
      for (const name of registry.listNames()) {
        const meta = TOOL_META[name];
        if (!meta) {
          log.warn({ event: 'mcp_tool_missing_meta', tool_name: name });
          continue;
        }
        mcp.tool(name, meta.description, meta.shape, async (args) => {
          // Pitfall 8: chat/iOS clients get disjoint idempotency key space.
          const synth = {
            ...(args as Record<string, unknown>),
            call_id: (args as any)?.call_id ?? `chat-${randomUUID()}`,
            turn_id: (args as any)?.turn_id ?? `chat-${Date.now()}-${randomUUID().slice(0, 8)}`,
          };
          const result = await registry.invoke(name, synth);
          return { content: [{ type: 'text' as const, text: JSON.stringify(result) }] };
        });
      }
      // MUST come BEFORE connect() per Anti-Pattern in RESEARCH.md.
      mcp.server.registerCapabilities({ tools: { listChanged: false } });

      const transport = new StreamableHTTPServerTransport({
        sessionIdGenerator: () => randomUUID(),
        onsessioninitialized: (sid) => {
          const now = Date.now();
          sessions.set(sid, { server: mcp, transport, sessionId: sid, createdAt: now, lastActivity: now });
          log.info({ event: 'mcp_session_opened', sid: sid.slice(0, 8) }, 'MCP session opened');
        },
      });
      transport.onclose = () => {
        const sid = transport.sessionId;
        if (sid && sessions.delete(sid)) {
          log.info({ event: 'mcp_session_closed', sid: sid.slice(0, 8), reason: 'transport_close' });
        }
      };
      await mcp.connect(transport);
      return { server: mcp, transport, sessionId: transport.sessionId ?? '', createdAt: Date.now(), lastActivity: Date.now() };
    }
    ```

    Step 6 — rewrite `app.all('/')` handler per Pattern 1 (lines 258-276):
    ```typescript
    app.all('/', async (req, res) => {
      const sid = req.header('Mcp-Session-Id');
      let session = sid ? sessions.get(sid) : undefined;
      if (session) {
        session.lastActivity = Date.now();
      } else {
        if (!isInitializeRequest(req.body)) {
          res.status(400).json({ jsonrpc: '2.0', error: { code: -32000, message: 'session_required' }, id: null });
          return;
        }
        if (sessions.size >= MCP_STREAM_MAX_SESSIONS) {
          log.warn({ event: 'mcp_session_cap_rejected', peer: req.socket.remoteAddress });
          res.status(503).json({ error: 'session_cap_reached' });
          return;
        }
        session = await createSession();
      }
      try {
        await session.transport.handleRequest(req, res, req.body);
      } catch (err) {
        log.warn({ event: 'mcp_stream_request_failed', err: err instanceof Error ? err.message : String(err) });
        if (!res.headersSent) res.status(500).json({ error: 'internal' });
      }
    });
    ```
