<!-- Google Distance Matrix API Shape -->
```
GET https://maps.googleapis.com/maps/api/distancematrix/json
  ?origins=München Hauptbahnhof
  &destinations=München Flughafen
  &mode=driving
  &key=AIza...
  [&departure_time=now]

Response: {
  status: "OK" | "INVALID_REQUEST" | "OVER_QUERY_LIMIT" | "REQUEST_DENIED" | "UNKNOWN_ERROR",
  origin_addresses: [string],
  destination_addresses: [string],
  rows: [{
    elements: [{
      status: "OK" | "NOT_FOUND" | "ZERO_RESULTS" | "MAX_ROUTE_LENGTH_EXCEEDED",
      duration: {value: number (sec), text: string},
      distance: {value: number (m), text: string},
      duration_in_traffic?: {value, text},  // nur bei departure_time+driving
    }]
  }]
}
```

Fuer uns: immer 1 origin + 1 destination → `rows[0].elements[0]`. `duration_in_traffic` nehmen wir als primary wenn vorhanden (genauer bei driving), sonst `duration`.

<!-- Client-Signatur -->
```typescript
export interface TravelTimeResult {
  duration_seconds: number
  distance_meters: number
  duration_text: string
  distance_text: string
  origin_resolved: string
  destination_resolved: string
  mode: string
  used_traffic: boolean  // true wenn duration_in_traffic verwendet
}

export interface GetTravelTimeOpts {
  origin: string
  destination: string
  mode?: 'driving' | 'walking' | 'bicycling' | 'transit'
  departureTime?: string  // 'now' | ISO; only applied to driving
  apiKey: string          // caller provides (aus config)
  timeoutMs?: number
  fetch?: typeof globalThis.fetch  // DI fuer Tests
}

export async function getTravelTime(opts: GetTravelTimeOpts): Promise<TravelTimeResult>
```

Errors: throws `MapsClientError(code, message)` wenn google_status != 'OK' oder element.status != 'OK' oder fetch-error/timeout. Handler catched und mapped zu `{ok:false, error}`.

<!-- Handler-Signatur (analog voice-check-calendar) -->
```typescript
export interface VoiceGetTravelTimeDeps {
  getTravelTime: (opts: GetTravelTimeOpts) => Promise<TravelTimeResult>  // DI (default imported above)
  apiKey: string  // from config.GOOGLE_MAPS_API_KEY
  jsonlPath?: string
  now?: () => number
  timeoutMs?: number
}
```
