const Schema = z.object({

  target_phone: z.string().regex(/^\+[1-9]\d{1,14}$/),  // E.164

  report_to_jid: z.string().min(1),
})

export interface VoiceRequestOutboundCallDeps {
  bridgeUrl: string         // http://10.0.0.2:4402
  bridgeAuthToken?: string  // optional bearer for /outbound
  fetch?: typeof globalThis.fetch
  jsonlPath?: string
  now?: () => number
}

// Handler body: zod-parse → POST bridgeUrl/outbound → map response.

<!-- Bridge /outbound request/response -->
```typescript
// Request:
POST /outbound
Content-Type: application/json
Authorization: Bearer <token if configured>
Body: {target_phone, goal, context, report_to_jid, call_id?}

// Response 200:
{outbound_task_id: uuid, estimated_start_ts: ISO, queue_position: int, status: 'queued'|'active'}

// Response 400: {error: 'bad_request', field, message}
// Response 401: {error: 'unauthorized'}
// Response 429: {error: 'queue_full'} (v1: queue_max=10)
```

<!-- OutboundTask -->
```typescript
interface OutboundTask {
  task_id: string
  target_phone: string
  goal: string
  context: string
  report_to_jid: string
  created_at: number
  started_at?: number
  ended_at?: number
  call_id?: string   // OpenAI-side
  status: 'queued' | 'active' | 'done' | 'failed' | 'escalated'
  error?: string
}
```

<!-- Outbound-Router lifecycle -->
```
enqueue(req): OutboundTask
  check max-queue-size
  push
  if idle: triggerExecute()
  return task

triggerExecute(): void
  if queue empty || activeCall → return
  pop next
  task.status='active', started_at=now
  openai.realtime.calls.create({...})
  start max-duration-timer (10min → end call)
  start 10min-escalation-timer (was schon active, verhindert task-escalation)

onCallEnd(callId, reason):
  find task by callId, mark done/failed
  report-back via core-mcp-client.sendDiscordMessage to report_to_jid
  triggerExecute() fuer naechsten Task

on10minEscalation(task):
  task.status='escalated'
  remove from queue
  post Discord alert to report_to_jid
```
