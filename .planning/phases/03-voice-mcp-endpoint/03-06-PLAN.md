<!-- flat-db-reader -->
```typescript
interface CacheEntry<T> { content: T; mtimeMs: number }

export async function readFlatDb<T>(path: string, defaultValue: T): Promise<T> {
  try {
    const stat = await fs.stat(path)
    const cached = CACHE.get(path) as CacheEntry<T> | undefined
    if (cached && cached.mtimeMs === stat.mtimeMs) return cached.content
    const text = await fs.readFile(path, 'utf8')
    const parsed = JSON.parse(text) as T
    CACHE.set(path, { content: parsed, mtimeMs: stat.mtimeMs })
    return parsed
  } catch (err) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') throw new FlatDbNotFound(path)
    if (err instanceof SyntaxError) throw new FlatDbParseError(path, err.message)
    throw err
  }
}
```
