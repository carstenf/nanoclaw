<!-- Tool-Registry Minimal-Shape -->
```typescript
export type ToolHandler = (args: unknown) => Promise<unknown>;
export interface ToolEntry { name: string; handler: ToolHandler; }
export class ToolRegistry {
  private tools = new Map<string, ToolHandler>();
  register(name: string, handler: ToolHandler): void { this.tools.set(name, handler) }
  async invoke(name: string, args: unknown): Promise<unknown> {
    const h = this.tools.get(name);
    if (!h) { const e: any = new Error(`unknown_tool:${name}`); e.code = 'unknown_tool'; throw e; }
    return h(args);
  }
  has(name: string): boolean { return this.tools.has(name) }
}
```
