```typescript
interface SkillLoadResult {
  exists: boolean
  body: string | null
  path: string
}

const SKILL_CACHE = new Map<string, {mtimeMs: number; body: string}>()

export async function loadSkill(topic: string, opts?: {skillsDir?: string, fs?: ...}): Promise<SkillLoadResult> {
  const skillsDir = opts?.skillsDir ?? SKILLS_DIR
  const path = `${skillsDir}/ask-core-${topic}/SKILL.md`
  try {
    const stat = await fs.stat(path)
    const cached = SKILL_CACHE.get(path)
    if (cached && cached.mtimeMs === stat.mtimeMs) {
      return {exists: true, body: cached.body, path}
    }
    const body = await fs.readFile(path, 'utf8')
    SKILL_CACHE.set(path, {mtimeMs: stat.mtimeMs, body})
    return {exists: true, body, path}
  } catch (err) {
    if ((err as NodeJS.ErrnoException).code === 'ENOENT') {
      return {exists: false, body: null, path}
    }
    throw err
  }
}
```
