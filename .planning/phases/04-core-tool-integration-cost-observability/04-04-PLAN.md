    ```bash
    #!/usr/bin/env bash
    # scripts/audit-audio.sh
    # REQ-QUAL-04 / REQ-LEGAL-03: monthly §201 audio-persistence audit.
    # Exits non-zero + Discord alert on ANY audio file found.
    # Runs on BOTH Lenovo1 (carsten_bot) and Hetzner (carsten, staggered +30 min).
    set -euo pipefail

    HOST=$(hostname -s)
    TS=$(date -u +%Y-%m-%dT%H:%M:%SZ)
    ROOTS=("$HOME" "/tmp" "/var/tmp")
    # Hetzner-only extra root: FreeSWITCH recordings
    if [ -d "/usr/local/freeswitch/recordings" ]; then
      ROOTS+=("/usr/local/freeswitch/recordings")
    fi

    FINDINGS=$(mktemp -t audit-audio.XXXXXX)
    trap 'rm -f "$FINDINGS"' EXIT

    for r in "${ROOTS[@]}"; do
      [ -d "$r" ] || continue
      # Pitfall 4: explicitly skip our own log output dirs so audit doesn't cry wolf
      find "$r" -type f \
        -not -path "$HOME/.local/share/systemd/*" \
        -not -path "/tmp/audit-audio.*" \
        \( -name "*.wav" -o -name "*.mp3" -o -name "*.opus" -o -name "*.flac" \) \
        2>/dev/null >> "$FINDINGS" || true
    done

    COUNT=$(wc -l < "$FINDINGS" | tr -d ' ')
    DISCORD_URL="${DISCORD_AUDIT_WEBHOOK_URL:-}"

    if [ -n "$DISCORD_URL" ]; then
      if [ "$COUNT" -gt 0 ]; then
        FILES_LIST=$(head -20 "$FINDINGS" | awk '{printf "• %s\n", $0}')
        MSG="$(printf '🚨 §201 AUDIT FAIL on %s — %s files found (first 20):\n```%s```' "$HOST" "$COUNT" "$FILES_LIST")"
      else
        MSG="$(printf '✅ §201 audit pass on %s (0 audio files) at %s' "$HOST" "$TS")"
      fi
      curl -fsS --max-time 5 -X POST "$DISCORD_URL" \
        -H "Content-Type: application/json" \
        -d "$(jq -n --arg c "$MSG" '{content: $c}')" || true
    fi

    # Exit-1 on any hit so systemd marks unit failed + journalctl captures
    if [ "$COUNT" -gt 0 ]; then
      echo "AUDIT FAIL: $COUNT files found" >&2
      cat "$FINDINGS" >&2
      exit 1
    fi
    echo "AUDIT PASS: 0 files"
    exit 0
    ```

    **Subtask 2a: `scripts/pricing-refresh.sh`:**
    ```bash
    #!/usr/bin/env bash
    # scripts/pricing-refresh.sh — Hetzner daily 02:00
    # Scrapes OpenAI gpt-realtime-mini pricing, writes voice_price_snapshots row via Core MCP,
    # alerts Discord on >5% drift vs pinned constants. NEVER auto-updates code.
    set -euo pipefail

    SOURCE_URL="${OPENAI_PRICING_SOURCE_URL:-https://platform.openai.com/docs/models/gpt-realtime-mini}"
    CORE_BASE="${CORE_MCP_BASE_URL:-http://10.0.0.2:3200}"
    CORE_TOKEN="${CORE_MCP_TOKEN:-}"
    DISCORD_URL="${DISCORD_AUDIT_WEBHOOK_URL:-}"
    STATE_REPO="${HOME}/nanoclaw-state"
    PRICING_JSON="${STATE_REPO}/voice-pricing.json"
    DRIFT_THRESHOLD="${PRICING_DRIFT_THRESHOLD:-0.05}"
    TS=$(date -u +%Y-%m-%dT%H:%M:%SZ)

    # Step 1: fetch source HTML
    HTML=$(mktemp -t pricing.XXXXXX.html)
    trap 'rm -f "$HTML"' EXIT
    if ! curl -fsS --max-time 15 "$SOURCE_URL" -o "$HTML"; then
      if [ -n "$DISCORD_URL" ]; then
        curl -fsS -X POST "$DISCORD_URL" -H "Content-Type: application/json" \
          -d "$(jq -n --arg c "⚠️ pricing-refresh: source unreachable ($SOURCE_URL) — manual check" '{content: $c}')" || true
      fi
      exit 0   # non-fatal per Pitfall 5; manual bump anyway
    fi

    # Step 2: parse pricing block. OpenAI docs inline pricing table — hunt with grep.
    # Known shape example:  "Audio input   $10.00 / 1M tokens"
    # Fallback: if parse fails, alert + exit 0. NEVER bogus values.
    parse_price() {
      local label="$1"
      grep -oE "${label}[^\\\$]*\\\$[0-9]+\\.[0-9]+" "$HTML" | head -1 | grep -oE '\\\$[0-9]+\\.[0-9]+' | tr -d '\\\$' | head -1
    }

    AUDIO_IN=$(parse_price "Audio input")
    AUDIO_OUT=$(parse_price "Audio output")
    AUDIO_CACHED=$(parse_price "Cached (audio input|input audio)")
    TEXT_IN=$(parse_price "Text input")
    TEXT_OUT=$(parse_price "Text output")

    if [ -z "$AUDIO_IN" ] || [ -z "$AUDIO_OUT" ]; then
      if [ -n "$DISCORD_URL" ]; then
        curl -fsS -X POST "$DISCORD_URL" -H "Content-Type: application/json" \
          -d "$(jq -n --arg c "⚠️ pricing-refresh: parse failed (audio_in=$AUDIO_IN audio_out=$AUDIO_OUT) — Carsten review docs" '{content: $c}')" || true
      fi
      exit 0
    fi

    # Step 3: compare vs pinned. Use AUDIO_IN as drift signal (largest cost contributor).
    ```

    ```typescript
    // src/drift-monitor.ts — Phase 4 (QUAL-03)
    // Rolling 24h P50 scan of turns-*.jsonl files.
    // Event: voice-bridge/src/turn-timing.ts produces {ts_iso, call_id, turn_id, t0_vad_end_ms, t2, t4, barge_in}.
    // Latency metric: t4 - t0 (first-tts-audio minus VAD-end).
    import fs from 'node:fs'; import path from 'node:path'; import os from 'node:os';
    import { logger } from './logger.js';

    export interface DriftMonitorDeps {
      baseDir?: string;  // default: process.env.BRIDGE_LOG_DIR || ~/nanoclaw/voice-container/runs
      now?: () => number;
      sendDiscordAlert: (msg: string) => Promise<void>;
      p50Threshold?: number;  // default 1200 (ms)
      windowMs?: number;      // default 24*60*60*1000
    }
    export interface DriftMonitorResult { samples: number; p50_ms: number; alerted: boolean }

    export function computeP50RollingWindow(
      baseDir: string, nowMs: number, windowMs: number,
    ): { samples: number[]; p50_ms: number } {
      const files = fs.existsSync(baseDir) ? fs.readdirSync(baseDir).filter(f => f.startsWith('turns-') && f.endsWith('.jsonl')) : [];
      const latencies: number[] = [];
      const cutoff = nowMs - windowMs;
      for (const file of files) {
        let raw = '';
        try { raw = fs.readFileSync(path.join(baseDir, file), 'utf8'); } catch { continue; }
        for (const line of raw.split('\n')) {
          if (!line.trim()) continue;
          try {
            const entry = JSON.parse(line);  // Pitfall 9: tolerate unknown fields
            if (typeof entry?.t0_vad_end_ms !== 'number') continue;
            if (typeof entry?.t4_first_tts_audio_ms !== 'number') continue;
            const tsIso = entry?.ts_iso ?? entry?.ts;
            const tsMs = tsIso ? Date.parse(String(tsIso)) : Number(entry?.ts ?? 0);
            if (!tsMs || tsMs < cutoff) continue;
            const lat = entry.t4_first_tts_audio_ms - entry.t0_vad_end_ms;
            if (lat > 0) latencies.push(lat);
          } catch { /* ignore */ }
        }
      }
      latencies.sort((a, b) => a - b);
      const p50 = latencies.length === 0 ? 0 : latencies[Math.floor(latencies.length / 2)];
      return { samples: latencies, p50_ms: p50 };
    }

    export async function runDriftMonitor(deps: DriftMonitorDeps): Promise<DriftMonitorResult> {
      const baseDir = deps.baseDir ?? process.env.BRIDGE_LOG_DIR ?? path.join(os.homedir(), 'nanoclaw', 'voice-container', 'runs');
      const now = deps.now ?? Date.now;
      const threshold = deps.p50Threshold ?? 1200;
      const windowMs = deps.windowMs ?? 24 * 60 * 60 * 1000;

      const { samples, p50_ms } = computeP50RollingWindow(baseDir, now(), windowMs);
      logger.info({ event: 'drift_monitor_run', samples: samples.length, p50_ms, threshold });

      const alerted = samples.length >= 10 && p50_ms > threshold;
      if (alerted) {
        await deps.sendDiscordAlert(`⚠️ Drift: P50 turn-latency rolling-24h = ${p50_ms}ms (threshold ${threshold}ms, samples=${samples.length})`);
      }
      return { samples: samples.length, p50_ms, alerted };
    }
    ```
