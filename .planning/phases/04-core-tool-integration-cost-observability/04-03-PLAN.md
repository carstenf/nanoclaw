    ```typescript
    import fs from 'fs'; import path from 'path';
    import { z } from 'zod';
    import { DATA_DIR } from '../config.js';
    import { logger } from '../logger.js';
    import { BadRequestError } from './voice-on-transcript-turn.js';
    import type { ToolHandler } from './index.js';

    const Schema = z.object({
      category: z.string().min(1).max(64),
      criteria: z.string().min(1).max(512),
      call_id: z.string().optional(),
    });

    export interface VoiceSearchCompetitorsDeps {
      /** Calls Claude with web-search tool enabled; returns offers[] JSON. */
      askCompetitorsBackend?: (category: string, criteria: string) => Promise<{ offers: Array<{ provider: string; price: string; terms: string; source_url: string }> }>;
      provider?: string;  // 'claude_web' | 'brave' | 'not_configured' | undefined
      jsonlPath?: string;
      now?: () => number;
    }

    export function makeVoiceSearchCompetitors(deps: VoiceSearchCompetitorsDeps): ToolHandler {
      const now = deps.now ?? Date.now;
      const jsonlPath = deps.jsonlPath ?? `${DATA_DIR}/voice-lookup.jsonl`;
      const provider = deps.provider ?? process.env.SEARCH_COMPETITORS_PROVIDER ?? '';

      return async function voiceSearchCompetitors(args: unknown): Promise<unknown> {
        const start = now();
        const parse = Schema.safeParse(args);
        if (!parse.success) {
          const f = parse.error.issues[0];
          throw new BadRequestError(String(f?.path?.[0] ?? 'input'), f?.message ?? 'invalid');
        }

        if (!provider || provider === 'not_configured') {
          logger.warn({ event: 'voice_search_competitors_not_configured' });
          appendJsonl(jsonlPath, { ts: new Date().toISOString(), event: 'search_competitors_not_configured', call_id: parse.data.call_id ?? null, latency_ms: now() - start });
          return { ok: false, error: 'not_configured' };
        }

        if (!deps.askCompetitorsBackend) {
          logger.warn({ event: 'voice_search_competitors_backend_missing', provider });
          return { ok: false, error: 'not_configured' };
        }

        try {
          const res = await deps.askCompetitorsBackend(parse.data.category, parse.data.criteria);
          appendJsonl(jsonlPath, {
            ts: new Date().toISOString(), event: 'search_competitors_done',
            tool: 'voice.search_competitors', category: parse.data.category,
            call_id: parse.data.call_id ?? null,
            offer_count: res.offers.length, provider, latency_ms: now() - start,
          });
          return { ok: true, result: { offers: res.offers } };
        } catch (e: unknown) {
          logger.warn({ event: 'voice_search_competitors_fail', err: (e as Error).message, provider });
          return { ok: false, error: 'backend_error' };
        }
      };
    }

    function appendJsonl(filePath: string, entry: object): void {
      try { fs.mkdirSync(path.dirname(filePath), { recursive: true }); fs.appendFileSync(filePath, JSON.stringify(entry) + '\n'); } catch {}
    }
    ```

    // src/mcp-stream-server.ts
    // Phase 4 (AC-07): StreamableHTTP MCP transport for Chat-Claude.
    // Port 3201 (port 3200 stays for Bridge REST fassade). Same ToolRegistry instance.
    // Pitfall 8: synthesize call_id='chat-<uuid>', turn_id='chat-<uuid>' to avoid
    // idempotency-key collision with live voice calls.
    import crypto from 'node:crypto';
    import express from 'express';
    import type { Request, Response } from 'express';
    import { McpServer } from '@modelcontextprotocol/sdk/server/mcp.js';
    import { StreamableHTTPServerTransport } from '@modelcontextprotocol/sdk/server/streamableHttp.js';
    import type { Server } from 'node:http';
    import { logger } from './logger.js';
    import { peerAllowlistMiddleware } from './peer-allowlist.js';
    import { MCP_STREAM_PORT, MCP_STREAM_BIND, MCP_STREAM_BEARER } from './config.js';
    import type { ToolRegistry } from './mcp-tools/index.js';

    const DEFAULT_ALLOWLIST = ['10.0.0.1', '10.0.0.2', '10.0.0.4', '10.0.0.5'];

    export interface McpStreamDeps {
      registry: ToolRegistry;
      bearerToken?: string;
      allowlist?: string[];
      log?: typeof logger;
    }

    export function buildMcpStreamApp(deps: McpStreamDeps): express.Application {
      const bearer = deps.bearerToken ?? MCP_STREAM_BEARER;
      if (!bearer) throw new Error('mcp-stream: MCP_STREAM_BEARER must be set');
      const log = deps.log ?? logger;
      const allowlist = deps.allowlist ?? DEFAULT_ALLOWLIST;

      const app = express();
      app.disable('x-powered-by');
      app.use(express.json({ limit: '2mb' }));

      // Health endpoint BEFORE auth — for Claude Chat discovery
      app.get('/mcp/stream/health', (_req: Request, res: Response) => {
        res.status(200).json({ ok: true, ts: Date.now(), bound_to: MCP_STREAM_BIND, tools: deps.registry.listNames() });
      });

      // Bearer auth middleware (cheap check first)
      app.use((req: Request, res: Response, next) => {
        if (req.path === '/mcp/stream/health') return next();
        const h = req.header('Authorization') ?? '';
        if (h !== `Bearer ${bearer}`) {
          log.warn({ event: 'mcp_stream_auth_fail', peer: req.socket.remoteAddress });
          res.status(401).json({ error: 'unauthorized' });
          return;
        }
        next();
      });

      // Peer-allowlist (Pitfall 6 — never expose public)
      app.use(peerAllowlistMiddleware(allowlist, log));

      // Mount MCP server with Pitfall-8-safe handler wrappers
      const mcp = new McpServer({ name: 'nanoclaw-voice', version: '1.0.0' });
      for (const name of deps.registry.listNames()) {
        const schema = deps.registry.schemaOf(name);
        mcp.tool(name, schema as any, async (args: unknown) => {
          // Pitfall 8: synthesize disjoint call_id/turn_id so idempotency cache
          // does NOT collide with live voice calls.
          const synthetic = {
            ...(args as object),
            call_id: `chat-${crypto.randomUUID()}`,
            turn_id: `chat-${Date.now()}-${crypto.randomUUID().slice(0, 8)}`,
          };
          const result = await deps.registry.invoke(name, synthetic);
          return { content: [{ type: 'text', text: JSON.stringify(result) }] };
        });
      }

      const transport = new StreamableHTTPServerTransport({ sessionIdGenerator: () => crypto.randomUUID() });
      // Connect synchronously — app.all handler calls transport.handleRequest
      void mcp.connect(transport);

      app.all('/mcp/stream', async (req: Request, res: Response) => {
        try {
          await transport.handleRequest(req, res, req.body);
        } catch (e: unknown) {
          log.warn({ event: 'mcp_stream_request_fail', err: (e as Error).message });
          if (!res.headersSent) res.status(500).json({ error: 'internal' });
        }
      });

      return app;
    }

    export function startMcpStreamServer(deps: McpStreamDeps): Server | null {
      if (!MCP_STREAM_BEARER) {
        logger.warn({ event: 'mcp_stream_server_not_started', reason: 'MCP_STREAM_BEARER unset — skipping StreamableHTTP startup' });
        return null;
      }
      const app = buildMcpStreamApp(deps);
      const server = app.listen(MCP_STREAM_PORT, MCP_STREAM_BIND, () => {
        logger.info({ event: 'mcp_stream_server_started', bind: MCP_STREAM_BIND, port: MCP_STREAM_PORT, tools: deps.registry.listNames() },
          `MCP Stream server listening on ${MCP_STREAM_BIND}:${MCP_STREAM_PORT}`);
      });
      server.on('error', (err: NodeJS.ErrnoException) => {
        if (err.code === 'EADDRINUSE') {
          logger.fatal({ event: 'mcp_stream_bind_failed', bind: MCP_STREAM_BIND, port: MCP_STREAM_PORT }, 'MCP stream server cannot bind — port in use');
          process.exit(1);
        }
        logger.error({ event: 'mcp_stream_server_error', err });
      });
      return server;
    }
