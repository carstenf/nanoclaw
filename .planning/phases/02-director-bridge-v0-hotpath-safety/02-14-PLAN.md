export async function emitFillerPhrase(
  ws: WebSocket,
  toolName: string,
  callId: string,
  log: Logger,
): Promise<boolean>

// Internal: synthetic conversation.item.create with assistant-message + response.create.
// Returns true if emitted, false if skipped (tool not in FILLER_PHRASE_TOOLS).
// Never throws — warn-logs on ws.send error, returns false.

Extensible via env `FILLER_PHRASE_TOOLS=ask_core,other_tool`. Per-tool message hard-coded in filler-inject.ts for v1 (simple Map<string, string>).

  )
}
