    ```typescript
    // voice-bridge/tests/replay/text-similarity.ts
    // D-31: Text similarity helper. Prefers SBERT cosine via @xenova/transformers;
    // falls back to dice-coefficient (from Plan 02-04) when SBERT is unavailable
    // or slow-to-load. Caches model instance across calls for CI re-use.
    // Emits a single init log `similarity=sbert` or `similarity=dice` so CI
    // output surfaces which implementation the gate ran under (Warning 3).
    import { diceCoefficient } from '../../src/readback/validator.js'

    let choice: 'sbert' | 'dice' | null = null
    let sbertReady: Promise<((s: string) => Promise<Float32Array>) | null> | null = null

    async function loadSbert(): Promise<((s: string) => Promise<Float32Array>) | null> {
      try {
        // eslint-disable-next-line @typescript-eslint/ban-ts-comment
        // @ts-expect-error optional dep
        const mod = await import('@xenova/transformers')
        const pipeline = (mod as { pipeline: unknown }).pipeline as
          (task: string, model: string, opts?: Record<string, unknown>) => Promise<
            (s: string, opts?: Record<string, unknown>) => Promise<{ data: Float32Array }>
          >
        const embed = await pipeline('feature-extraction',
          'Xenova/all-MiniLM-L6-v2', { quantized: true })
        return async (s: string) => {
          const out = await embed(s, { pooling: 'mean', normalize: true })
          return out.data
        }
      } catch {
        return null
      }
    }

    function cosine(a: Float32Array, b: Float32Array): number {
      let dot = 0, na = 0, nb = 0
      for (let i = 0; i < a.length; i++) { dot += a[i] * b[i]; na += a[i] ** 2; nb += b[i] ** 2 }
      return dot / Math.max(Math.sqrt(na) * Math.sqrt(nb), 1e-9)
    }

    async function announce(): Promise<void> {
      if (choice !== null) return
      if (!sbertReady) sbertReady = loadSbert()
      const embed = await sbertReady
      choice = embed ? 'sbert' : 'dice'
      // Single init-time CI-visible line (Warning 3 fix — makes fallback observable).
      // Use console.log NOT the pino bridge logger — this is test infrastructure
      // output destined for CI stdout, not a structured JSONL event.
      console.log(`similarity=${choice}`)
    }

    export async function cosineOrDice(a: string, b: string): Promise<number> {
      await announce()
      if (choice === 'sbert') {
        const embed = await sbertReady
        if (embed) {
          const [ea, eb] = await Promise.all([embed(a), embed(b)])
          return cosine(ea, eb)
        }
      }
      return diceCoefficient(a, b)
    }

    export function getSimilarityChoice(): 'sbert' | 'dice' | null {
      return choice
    }

    export const TEXT_SIMILARITY_MIN = 0.80
    ```

    ```typescript
        .trim().split('\n').map((l) => JSON.parse(l) as FixtureTurn)
      const golden = JSON.parse(readFileSync(GOLDEN_PATH, 'utf8'))[name] as FixtureGolden
      return { turns, golden }
    }

    export function mockLog(): pino.Logger {
      return { info: vi.fn(), warn: vi.fn(), error: vi.fn(),
        debug: vi.fn(), trace: vi.fn(), fatal: vi.fn() } as unknown as pino.Logger
    }

    export interface TurnMeasurement {
      turn_idx: number
      /** Wall-clock Bridge-side work for tool turns; fixture-reported for non-tool turns. */
      elapsed_ms: number
      /** True if this turn invoked a tool (counted for p50/p95 bucket). */
      counted_for_percentile: boolean
      golden_ms: number
      in_band: boolean
    }

    export async function runReplayAgainstBridge(
      fixtureName: string,
    ): Promise<TurnMeasurement[]> {
      const { turns, golden } = loadFixture(fixtureName)
      clearCall('replay')
      const out: TurnMeasurement[] = []
      const log = mockLog()
      for (const turn of turns) {
        const goldenTurn = golden.turns[turn.turn_idx]
        const golden_ms = goldenTurn?.t_first_audio_ms ?? turn.t_first_audio_ms

        let elapsed: number
        let counted: boolean

        if (turn.tool_call_triggered && turn.tool_name) {
          // Bridge-side dispatch: measure actual elapsed wall-clock.
          const t0 = performance.now()
          const args = turn.tool_args_str ? JSON.parse(turn.tool_args_str) : {}
          await invokeIdempotent(
            'replay', `turn-${turn.turn_idx}`, turn.tool_name, args,
            async () => dispatchTool('replay', `turn-${turn.turn_idx}`,
              turn.tool_name!, args, log),
            log,
          )
          elapsed = performance.now() - t0
          counted = true
        } else {
          // Non-tool turn: use platform-reported fixture t_first_audio_ms to
          // preserve the golden band contract, but DO NOT count toward p50/p95.
          elapsed = golden_ms
          counted = false
        }

        const in_band = Math.abs(elapsed - golden_ms) <= golden.tolerance_ms ||
          // Bridge-side dispatch is orders of magnitude faster than OpenAI
          // round-trip fixtures. For tool turns, accept elapsed < golden + tol.
          elapsed < golden_ms + golden.tolerance_ms

        out.push({
          turn_idx: turn.turn_idx,
          elapsed_ms: elapsed,
          counted_for_percentile: counted,
          golden_ms,
          in_band,
        })
      }
      return out
    }

    export function percentile(values: number[], p: number): number {
      if (values.length === 0) return 0
      const sorted = [...values].sort((a, b) => a - b)
      const idx = Math.min(sorted.length - 1, Math.floor(p * sorted.length))
      return sorted[idx]
    }
    ```
