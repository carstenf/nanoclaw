  /** Spawn / reach the per-call container-agent. */
  invokeAgent: (input: VoiceTriggersInitInput) => Promise<{ instructions: string }>;
  jsonlPath?: string;
  now?: () => number;

export function makeVoiceTriggersInit(deps: VoiceTriggersInitDeps): ToolHandler {
  const jsonlPath = deps.jsonlPath ?? path.join(DATA_DIR, 'voice-triggers.jsonl');
  const nowFn = deps.now ?? (() => Date.now());

### `src/voice-trigger-queue.ts` (NEW — utility, per-key FIFO)

**Analog:** `src/group-queue.ts` (per-`groupJid` queue impl). The closest existing per-key queue in the repo. Not a literal copy — `GroupQueue` is heavy (container-process-aware, retry-aware); we want a slim FIFO that wraps a `Map<call_id, Promise<unknown>>` chain.

**Per-key queue pattern (semantic) — from `group-queue.ts:30-130`:**

export class GroupQueue {
  private groups = new Map<string, GroupState>();
  ...
  enqueueTask(groupJid: string, taskId: string, fn: () => Promise<void>): void {
    const state = this.getGroup(groupJid);
    if (state.active) { state.pendingTasks.push({ id: taskId, groupJid, fn }); return; }
    this.runTask(groupJid, { id: taskId, groupJid, fn }).catch(...)
  }
  ...
  private drainGroup(groupJid: string): void {
    if (state.pendingTasks.length > 0) {
      const task = state.pendingTasks.shift()!;
      this.runTask(groupJid, task).catch(...)
      return;
    }
  }
}

**Recommended slim pattern for `voice-trigger-queue.ts` (Promise-chain per key):**
```typescript
export class VoiceTriggerQueue {
  private chains = new Map<string, Promise<unknown>>();
  private depths = new Map<string, number>();

  enqueue<T>(callId: string, fn: () => Promise<T>): Promise<T> {
    const prev = this.chains.get(callId) ?? Promise.resolve();
    this.depths.set(callId, (this.depths.get(callId) ?? 0) + 1);
    const next = prev.then(fn, fn);  // run on success or failure of prior
    this.chains.set(callId, next.finally(() => {
      this.depths.set(callId, Math.max(0, (this.depths.get(callId) ?? 1) - 1));
    }));
    return next;
  }

  /** D-11 garbage collection on end_call. */
  gc(callId: string): void {
    this.chains.delete(callId);
    this.depths.delete(callId);
  }

  /** Exposed via /health for observability per Specifics §FIFO queue invariants. */
  depth(callId: string): number {
    return this.depths.get(callId) ?? 0;
  }
}
```

**Deviations vs `group-queue.ts`:**
- No container-process-state (`active`, `process`, `groupFolder`) — pure Promise chain.
- No retry / shutdown — `gc()` is the only explicit removal path.
- No concurrency cap (one chain per call_id; calls are independent).
