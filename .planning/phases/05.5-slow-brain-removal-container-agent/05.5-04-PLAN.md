```typescript
onTranscriptTurn: async (turnId, transcript) => {
  const existing = router.getCall(callId);
  if (!existing) { /* warn + return */ return; }
  // AMD VAD-fallback unchanged — must remain in both modes:
  getAmdClassifier()?.onTranscript(transcript);
  if (REASONING_MODE === 'container-agent') {
    void existing.nanoclawMcp?.transcript({
      call_id: callId,
      turn_id: Number(turnId.replace(/^\D+/, '')) || 0,
      transcript: { turns: existing.turnHistory ?? [{ role: 'counterpart', text: transcript, started_at: new Date().toISOString() }] },
      fast_brain_state: {},
    }).then((res) => {
      if (res?.instructions_update) {
        updateInstructions(existing.sideband.state, res.instructions_update, log);
      }
    }).catch((err) => log.warn({ event: 'transcript_trigger_failed', call_id: callId, err: (err as Error).message }));
  } else {
    existing.slowBrain.push({ turnId, transcript });
  }
},
```
